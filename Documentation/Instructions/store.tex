\subsection{Store}
\begin{verbatim}
    st (BASE + IMM), ac    ; Direct
    st ((BASE + IMM)), ac  ; Deferred
    st (+(BASE + IMM)), ac ; Auto Increment
    st ((BASE + IMM)-), ac ; Auto Decrement
\end{verbatim}
